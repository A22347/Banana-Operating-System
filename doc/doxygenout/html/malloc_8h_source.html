45/*
46 malloc(size_t n)
47 Returns a pointer to a newly allocated chunk of at least n bytes, or
48 null if no space is available. Additionally, on failure, errno is
49 set to ENOMEM on ANSI C systems.
50
51 If n is zero, malloc returns a minimum-sized chunk. The minimum size
52 is 16 bytes on most 32bit systems, and either 24 or 32 bytes on
53 64bit systems, depending on internal size and alignment restrictions.
54
55 On most systems, size_t is an unsigned type. Calls with values of n
56 that appear "negative" when signed are interpreted as requests for
57 huge amounts of space, which will most often fail.
58
59 The maximum allowed value of n differs across systems, but is in all
60 cases less (typically by 8K) than the maximum representable value of
61 a size_t. Requests greater than this value result in failure.
62*/
63
64#ifndef USE_DL_PREFIX
65void* malloc(size_t);
66#else
67void* dlmalloc(size_t);
68#endif

98/*
99 realloc(void* p, size_t n)
100 Returns a pointer to a chunk of size n that contains the same data
101 as does chunk p up to the minimum of (n, p's size) bytes.
102
103 The returned pointer may or may not be the same as p. The algorithm
104 prefers extending p when possible, otherwise it employs the
105 equivalent of a malloc-copy-free sequence.
106
107 If p is null, realloc is equivalent to malloc.
108
109 If space is not available, realloc returns null, errno is set (if on
110 ANSI) and p is NOT freed.
111
112 if n is for fewer bytes than already held by p, the newly unused
113 space is lopped off and freed if possible. Unless the #define
114 REALLOC_ZERO_BYTES_FREES is set, realloc with a size argument of
115 zero (re)allocates a minimum-sized chunk.
116
117 Large chunks that were internally obtained via mmap will always
118 be reallocated using malloc-copy-free sequences unless
119 the system supports MREMAP (currently only linux).
120
121 The old unix realloc convention of allowing the last-free'd chunk
122 to be used as an argument to realloc is not supported.
123*/
124

126void* realloc(void*, size_t);
127#else
128void* dlrealloc(void*, size_t);

165/*
166 independent_calloc(size_t n_elements, size_t element_size, void* chunks[]);
167
168 independent_calloc is similar to calloc, but instead of returning a
169 single cleared space, it returns an array of pointers to n_elements
170 independent elements, each of which can hold contents of size
171 elem_size. Each element starts out cleared, and can be
172 independently freed, realloc'ed etc. The elements are guaranteed to
173 be adjacently allocated (this is not guaranteed to occur with
174 multiple callocs or mallocs), which may also improve cache locality
175 in some applications.
176
177 The "chunks" argument is optional (i.e., may be null, which is
178 probably the most typical usage). If it is null, the returned array
179 is itself dynamically allocated and should also be freed when it is
180 no longer needed. Otherwise, the chunks array must be of at least
181 n_elements in length. It is filled in with the pointers to the
182 chunks.
183
184 In either case, independent_calloc returns this pointer array, or
185 null if the allocation failed. If n_elements is zero and "chunks"
186 is null, it returns a chunk representing an array with zero elements
187 (which should be freed if not wanted).
188
189 Each element must be individually freed when it is no longer
190 needed. If you'd like to instead be able to free all at once, you
191 should instead use regular calloc and assign pointers into this
192 space to represent elements. (In this case though, you cannot
193 independently free elements.)
194
195 independent_calloc simplifies and speeds up implementations of many
196 kinds of pools. It may also be useful when constructing large data
197 structures that initially have a fixed number of fixed-sized nodes,
198 but the number is not known at compile time, and some of the nodes
199 may later need to be freed. For example:
200
201 struct Node { int item; struct Node* next; };
202
203 struct Node* build_list() {
204 struct Node** pool;
205 int n = read_number_of_nodes_needed();
206 if (n <= 0) return 0;
207 pool = (struct Node**)(independent_calloc(n, sizeof(struct Node), 0);
208 if (pool == 0) return 0; // failure
209 // organize into a linked list...
210 struct Node* first = pool[0];
211 for (i = 0; i < n-1; ++i)
212 pool[i]->next = pool[i+1];
213 free(pool); // Can now free the array (or not, if it is needed later)
214 return first;
215 }
216*/
217

219void** independent_calloc(size_t, size_t, void**);
220#else
221void** dlindependent_calloc(size_t, size_t, void**);
222#endif

224/*
225 independent_comalloc(size_t n_elements, size_t sizes[], void* chunks[]);
226
227 independent_comalloc allocates, all at once, a set of n_elements
228 chunks with sizes indicated in the "sizes" array. It returns
229 an array of pointers to these elements, each of which can be
230 independently freed, realloc'ed etc. The elements are guaranteed to
231 be adjacently allocated (this is not guaranteed to occur with
232 multiple callocs or mallocs), which may also improve cache locality
233 in some applications.
234
235 The "chunks" argument is optional (i.e., may be null). If it is null
236 the returned array is itself dynamically allocated and should also
237 be freed when it is no longer needed. Otherwise, the chunks array
238 must be of at least n_elements in length. It is filled in with the
239 pointers to the chunks.
240
241 In either case, independent_comalloc returns this pointer array, or
242 null if the allocation failed. If n_elements is zero and chunks is
243 null, it returns a chunk representing an array with zero elements
244 (which should be freed if not wanted).
245
246 Each element must be individually freed when it is no longer
247 needed. If you'd like to instead be able to free all at once, you
248 should instead use a single regular malloc, and assign pointers at
249 particular offsets in the aggregate space. (In this case though, you
250 cannot independently free elements.)
251
252 independent_comallac differs from independent_calloc in that each
253 element may have a different size, and also that it does not
254 automatically clear elements.
255
256 independent_comalloc can be used to speed up allocation in cases
257 where several structs or objects must always be allocated at the
258 same time. For example:
259
260 struct Head { ... }
261 struct Foot { ... }
262
263 void send_message(char* msg) {
264 int msglen = strlen(msg);
265 size_t sizes[3] = { sizeof(struct Head), msglen, sizeof(struct Foot) };
266 void* chunks[3];
267 if (independent_comalloc(3, sizes, chunks) == 0)
268 die();
269 struct Head* head = (struct Head*)(chunks[0]);
270 char* body = (char*)(chunks[1]);
271 struct Foot* foot = (struct Foot*)(chunks[2]);
272 // ...
273 }
274
275 In general though, independent_comalloc is worth using only for
276 larger values of n_elements. For small values, you probably won't
277 detect enough difference from series of malloc calls to bother.
278
279 Overuse of independent_comalloc can increase overall memory usage,
280 since it cannot reuse existing noncontiguous small chunks that
281 might be available for some of the elements.
282*/
283

399/*
400 mallinfo()
401 Returns (by copy) a struct containing various summary statistics:
402
403 arena: current total non-mmapped bytes allocated from system
404 ordblks: the number of free chunks
405 smblks: the number of fastbin blocks (i.e., small chunks that
406 have been freed but not use resused or consolidated)
407 hblks: current number of mmapped regions
408 hblkhd: total bytes held in mmapped regions
409 usmblks: the maximum total allocated space. This will be greater
410 than current total if trimming has occurred.
411 fsmblks: total bytes held in fastbin blocks
412 uordblks: current total allocated space (normal or mmapped)

438 mallopt(int parameter_number, int parameter_value)
439 Sets tunable parameters The format is to provide a
440 (parameter-number, parameter-value) pair. mallopt then sets the
441 corresponding parameter to the argument value if it can (i.e., so
442 long as the value is meaningful), and returns 1 if successful else
443 0. SVID/XPG defines four standard param numbers for mallopt,
444 normally defined in malloc.h. Only one of these (M_MXFAST) is used
445 in this malloc. The others (M_NLBLKS, M_GRAIN, M_KEEP) don't apply,
446 so setting them has no effect. But this malloc also supports four
447 other options in mallopt. See below for details. Briefly, supported
448 parameters are as follows (listed defaults are for "typical"
449 configurations).
450
451 Symbol param # default allowed param values
452 M_MXFAST 1 64 0-80 (0 disables fastbins)
453 M_TRIM_THRESHOLD -1 128*1024 any (-1U disables trimming)
454 M_TOP_PAD -2 0 any
455 M_MMAP_THRESHOLD -3 128*1024 any (or 0 if no MMAP support)
456 M_MMAP_MAX -4 65536 any (0 disables use of mmap)

465/* Descriptions of tuning options */
466
467/*
468 M_MXFAST is the maximum request size used for "fastbins", special bins
469 that hold returned chunks without consolidating their spaces. This
470 enables future requests for chunks of the same size to be handled
471 very quickly, but can increase fragmentation, and thus increase the
472 overall memory footprint of a program.
473
474 This malloc manages fastbins very conservatively yet still
475 efficiently, so fragmentation is rarely a problem for values less
476 than or equal to the default. The maximum supported value of MXFAST
477 is 80. You wouldn't want it any higher than this anyway. Fastbins
478 are designed especially for use with many small structs, objects or
479 strings -- the default handles structs/objects/arrays with sizes up
480 to 8 4byte fields, or small strings representing words, tokens,
481 etc. Using fastbins for larger objects normally worsens
482 fragmentation without improving speed.
483
484 You can reduce M_MXFAST to 0 to disable all use of fastbins. This
485 causes the malloc algorithm to be a closer approximation of
486 fifo-best-fit in all cases, not just for larger requests, but will
487 generally cause it to be slower.
488*/
489
490#ifndef M_MXFAST
491#define M_MXFAST 1
492#endif
493
494/*
495 M_TRIM_THRESHOLD is the maximum amount of unused top-most memory
496 to keep before releasing via malloc_trim in free().
497
498 Automatic trimming is mainly useful in long-lived programs.
499 Because trimming via sbrk can be slow on some systems, and can
500 sometimes be wasteful (in cases where programs immediately
501 afterward allocate more large chunks) the value should be high
502 enough so that your overall system performance would improve by
503 releasing this much memory.
504
505 The trim threshold and the mmap control parameters (see below)
506 can be traded off with one another. Trimming and mmapping are
507 two different ways of releasing unused memory back to the
508 system. Between these two, it is often possible to keep
509 system-level demands of a long-lived program down to a bare
510 minimum. For example, in one test suite of sessions measuring
511 the XF86 X server on Linux, using a trim threshold of 128K and a
512 mmap threshold of 192K led to near-minimal long term resource
513 consumption.
514
515 If you are using this malloc in a long-lived program, it should
516 pay to experiment with these values. As a rough guide, you
517 might set to a value close to the average size of a process
518 (program) running on your system. Releasing this much memory
519 would allow such a process to run in memory. Generally, it's
520 worth it to tune for trimming rather tham memory mapping when a
521 program undergoes phases where several large chunks are
522 allocated and released in ways that can reuse each other's
523 storage, perhaps mixed with phases where there are no such
524 chunks at all. And in well-behaved long-lived programs,
525 controlling release of large blocks via trimming versus mapping
526 is usually faster.
527
528 However, in most programs, these parameters serve mainly as
529 protection against the system-level effects of carrying around
530 massive amounts of unneeded memory. Since frequent calls to
531 sbrk, mmap, and munmap otherwise degrade performance, the default
532 parameters are set to relatively high values that serve only as
533 safeguards.
534
535 The trim value It must be greater than page size to have any useful
536 effect. To disable trimming completely, you can set to
537 (unsigned long)(-1)
538
539 Trim settings interact with fastbin (MXFAST) settings: Unless
540 compiled with TRIM_FASTBINS defined, automatic trimming never takes
541 place upon freeing a chunk with size less than or equal to
542 MXFAST. Trimming is instead delayed until subsequent freeing of
543 larger chunks. However, you can still force an attempted trim by
544 calling malloc_trim.
545
546 Also, trimming is not generally possible in cases where
547 the main arena is obtained via mmap.
548
549 Note that the trick some people use of mallocing a huge space and
550 then freeing it at program startup, in an attempt to reserve system
551 memory, doesn't have the intended effect under automatic trimming,
552 since that memory will immediately be returned to the system.
553*/
554
555#define M_TRIM_THRESHOLD -1
556
557/*
558 M_TOP_PAD is the amount of extra `padding' space to allocate or
559 retain whenever sbrk is called. It is used in two ways internally:
560
561 * When sbrk is called to extend the top of the arena to satisfy
562 a new malloc request, this much padding is added to the sbrk
563 request.
564
565 * When malloc_trim is called automatically from free(),
566 it is used as the `pad' argument.
567
568 In both cases, the actual amount of padding is rounded
569 so that the end of the arena is always a system page boundary.
570
571 The main reason for using padding is to avoid calling sbrk so
572 often. Having even a small pad greatly reduces the likelihood
573 that nearly every malloc request during program start-up (or
574 after trimming) will invoke sbrk, which needlessly wastes
575 time.
576
577 Automatic rounding-up to page-size units is normally sufficient
578 to avoid measurable overhead, so the default is 0. However, in
579 systems where sbrk is relatively slow, it can pay to increase
580 this value, at the expense of carrying around more memory than
581 the program needs.
582*/
583
584#define M_TOP_PAD -2

587/*
588 M_MMAP_THRESHOLD is the request size threshold for using mmap()
589 to service a request. Requests of at least this size that cannot
590 be allocated using already-existing space will be serviced via mmap.
591 (If enough normal freed space already exists it is used instead.)
592
593 Using mmap segregates relatively large chunks of memory so that
594 they can be individually obtained and released from the host
595 system. A request serviced through mmap is never reused by any
596 other request (at least not directly; the system may just so
597 happen to remap successive requests to the same locations).
598
599 Segregating space in this way has the benefits that:
600
601 1. Mmapped space can ALWAYS be individually released back
602 to the system, which helps keep the system level memory
603 demands of a long-lived program low.
604 2. Mapped memory can never become `locked' between
605 other chunks, as can happen with normally allocated chunks, which
606 means that even trimming via malloc_trim would not release them.
607 3. On some systems with "holes" in address spaces, mmap can obtain
608 memory that sbrk cannot.
609
610 However, it has the disadvantages that:
611
612 1. The space cannot be reclaimed, consolidated, and then
613 used to service later requests, as happens with normal chunks.
614 2. It can lead to more wastage because of mmap page alignment
615 requirements
616 3. It causes malloc performance to be more dependent on host
617 system memory management support routines.
618
619 The advantages of mmap nearly always outweigh disadvantages for
620 "large" chunks, but the value of "large" varies across systems. The
621 default is an empirically derived value that works well in most
622 systems.
623*/
624
625#define M_MMAP_THRESHOLD -3
626
627/*
628 M_MMAP_MAX is the maximum number of requests to simultaneously
629 service using mmap. This parameter exists because
630 some systems have a limited number of internal tables for
631 use by mmap, and using more than a few of them may degrade
632 performance.
633
634 The default is set to a value that serves only as a safeguard.
635 Setting to 0 disables use of mmap for servicing large requests. If
636 mmap is not supported on a system, the default value is 0, and
637 attempts to set it to non-zero values in mallopt will fail.
638*/
639
640#define M_MMAP_MAX -4
