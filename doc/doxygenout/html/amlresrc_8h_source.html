174#define ACPI_RESTAG_ALIGNMENT "_ALN"

177#define ACPI_RESTAG_TYPESPECIFICATTRIBUTES "_ATT"
178#define ACPI_RESTAG_BASEADDRESS "_BAS"
179#define ACPI_RESTAG_BUSMASTER "_BM_" /* Master(1), Slave(0) */
180#define ACPI_RESTAG_DEBOUNCETIME "_DBT"
181#define ACPI_RESTAG_DECODE "_DEC"
182#define ACPI_RESTAG_DEVICEPOLARITY "_DPL"
183#define ACPI_RESTAG_DMA "_DMA"
184#define ACPI_RESTAG_DMATYPE "_TYP" /* Compatible(0), A(1), B(2), F(3) */
185#define ACPI_RESTAG_DRIVESTRENGTH "_DRS"
186#define ACPI_RESTAG_ENDIANNESS "_END"
187#define ACPI_RESTAG_FLOWCONTROL "_FLC"
188#define ACPI_RESTAG_FUNCTION "_FUN"
189#define ACPI_RESTAG_GRANULARITY "_GRA"
190#define ACPI_RESTAG_INTERRUPT "_INT"
191#define ACPI_RESTAG_INTERRUPTLEVEL "_LL_" /* ActiveLo(1), ActiveHi(0) */
192#define ACPI_RESTAG_INTERRUPTSHARE "_SHR" /* Shareable(1), NoShare(0) */
193#define ACPI_RESTAG_INTERRUPTTYPE "_HE_" /* Edge(1), Level(0) */
194#define ACPI_RESTAG_IORESTRICTION "_IOR"
195#define ACPI_RESTAG_LENGTH "_LEN"
196#define ACPI_RESTAG_LINE "_LIN"
197#define ACPI_RESTAG_MEMATTRIBUTES "_MTP" /* Memory(0), Reserved(1), ACPI(2), NVS(3) */
198#define ACPI_RESTAG_MEMTYPE "_MEM" /* NonCache(0), Cacheable(1) Cache+combine(2), Cache+prefetch(3) */
199#define ACPI_RESTAG_MAXADDR "_MAX"
200#define ACPI_RESTAG_MINADDR "_MIN"
201#define ACPI_RESTAG_MAXTYPE "_MAF"
202#define ACPI_RESTAG_MINTYPE "_MIF"
203#define ACPI_RESTAG_MODE "_MOD"
204#define ACPI_RESTAG_PARITY "_PAR"
205#define ACPI_RESTAG_PHASE "_PHA"
206#define ACPI_RESTAG_PIN "_PIN"
207#define ACPI_RESTAG_PINCONFIG "_PPI"
208#define ACPI_RESTAG_PINCONFIG_TYPE "_TYP"
209#define ACPI_RESTAG_PINCONFIG_VALUE "_VAL"
210#define ACPI_RESTAG_POLARITY "_POL"
211#define ACPI_RESTAG_REGISTERBITOFFSET "_RBO"
212#define ACPI_RESTAG_REGISTERBITWIDTH "_RBW"
213#define ACPI_RESTAG_RANGETYPE "_RNG"
214#define ACPI_RESTAG_READWRITETYPE "_RW_" /* ReadOnly(0), Writeable (1) */
215#define ACPI_RESTAG_LENGTH_RX "_RXL"
216#define ACPI_RESTAG_LENGTH_TX "_TXL"
217#define ACPI_RESTAG_SLAVEMODE "_SLV"

220#define ACPI_RESTAG_TRANSLATION "_TRA"
221#define ACPI_RESTAG_TRANSTYPE "_TRS" /* Sparse(1), Dense(0) */
