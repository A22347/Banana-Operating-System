165 * Walk state - current state of a parse tree walk. Used for both a leisurely
166 * stroll through the tree (for whatever reason), and for control method
167 * execution.
168 */

322
323 const ACPI_PREDEFINED_INFO *Predefined; /* Used if Node is a predefined name */

331 UINT8 PassNumber; /* Parser pass number */
